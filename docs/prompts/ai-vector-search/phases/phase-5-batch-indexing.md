# Phase 5: Batch Indexing

> **Goal:** Index existing tests for semantic search
> **Value Delivered:** All historical tests become searchable by intent
> **Dependencies:** Phase 1 (embedding utilities)
> **Estimated Steps:** 4

---

## Overview

This phase prepares the foundation for semantic test search:
1. Add test metadata embedding column
2. Create batch indexing script/API
3. Index test names, descriptions, and file paths
4. Create test search index table

### Step 5.1: Create Test Search Index Table

**File:** `scripts/016_add_test_search_index.sql`

```sql
-- Migration 016: Add Test Search Index for Semantic Search
-- Purpose: Enable natural language search for tests
-- Date: 2026-01-12

-- ============================================
-- Test Search Index Table
-- ============================================
-- Stores embeddings for test metadata (name, file, description)
-- This is separate from error_embedding which is for failures

CREATE TABLE IF NOT EXISTS test_search_index (
  id SERIAL PRIMARY KEY,
  organization_id INTEGER NOT NULL REFERENCES organizations(id) ON DELETE CASCADE,

  -- Test identification
  test_signature TEXT NOT NULL,     -- MD5(file::name)
  test_name TEXT NOT NULL,
  test_file TEXT NOT NULL,
  test_description TEXT,            -- Optional describe() block text

  -- Search embedding
  search_embedding vector(768),     -- Embedding of name + file + description

  -- Metadata
  last_seen_execution_id INTEGER REFERENCES test_executions(id),
  total_runs INTEGER DEFAULT 1,
  pass_rate FLOAT,
  last_status TEXT,
  last_indexed_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),

  -- Unique per org + signature
  UNIQUE(organization_id, test_signature)
);

-- Index for fast semantic search
CREATE INDEX IF NOT EXISTS idx_test_search_embedding_hnsw
ON test_search_index
USING hnsw (search_embedding vector_cosine_ops)
WHERE search_embedding IS NOT NULL;

-- Index for org filtering
CREATE INDEX IF NOT EXISTS idx_test_search_org
ON test_search_index(organization_id);

-- Index for text fallback search
CREATE INDEX IF NOT EXISTS idx_test_search_name_trgm
ON test_search_index
USING gin (test_name gin_trgm_ops);

CREATE INDEX IF NOT EXISTS idx_test_search_file_trgm
ON test_search_index
USING gin (test_file gin_trgm_ops);

-- Add comments
COMMENT ON TABLE test_search_index IS
  'Search index for semantic test discovery. Embeddings generated from test names, files, and descriptions.';

COMMENT ON COLUMN test_search_index.search_embedding IS
  'Vector embedding of test metadata for semantic search. Generated by Gemini text-embedding-004.';

-- ============================================
-- Helper function for semantic search
-- ============================================
CREATE OR REPLACE FUNCTION semantic_test_search(
  p_organization_id INTEGER,
  p_query_embedding vector(768),
  p_limit INTEGER DEFAULT 20
)
RETURNS TABLE (
  test_signature TEXT,
  test_name TEXT,
  test_file TEXT,
  similarity FLOAT,
  total_runs INTEGER,
  pass_rate FLOAT,
  last_status TEXT
)
LANGUAGE SQL
STABLE
AS $$
  SELECT
    test_signature,
    test_name,
    test_file,
    1 - (search_embedding <=> p_query_embedding) as similarity,
    total_runs,
    pass_rate,
    last_status
  FROM test_search_index
  WHERE organization_id = p_organization_id
    AND search_embedding IS NOT NULL
  ORDER BY search_embedding <=> p_query_embedding
  LIMIT p_limit;
$$;

COMMENT ON FUNCTION semantic_test_search IS
  'Perform semantic search on tests using vector similarity.';

-- ============================================
-- RLS Policies
-- ============================================
ALTER TABLE test_search_index ENABLE ROW LEVEL SECURITY;

-- View policy
CREATE POLICY "select_test_search_index" ON test_search_index FOR SELECT
  USING (
    is_org_member(organization_id)
    OR is_system_admin()
    OR is_service_account()
  );

-- Insert/Update policy (service account only)
CREATE POLICY "insert_test_search_index" ON test_search_index FOR INSERT
  WITH CHECK (is_service_account() OR is_system_admin());

CREATE POLICY "update_test_search_index" ON test_search_index FOR UPDATE
  USING (is_service_account() OR is_system_admin());
```

### Step 5.2: Create Test Index Service

**File:** `lib/services/test-indexing-service.ts`

```typescript
/**
 * Test Indexing Service
 *
 * Manages the test search index for semantic search.
 * Indexes test metadata (name, file, description) for discovery.
 */

import { getSql } from "@/lib/db/connection"
import { generateEmbedding, generateEmbeddingsBatch, toVectorString } from "@/lib/ai"

/**
 * Prepare test metadata for embedding
 */
function prepareTestMetadata(
  testName: string,
  testFile: string,
  testDescription?: string
): string {
  const parts = [
    `Test: ${testName}`,
    `File: ${testFile}`,
  ]

  if (testDescription) {
    parts.push(`Description: ${testDescription}`)
  }

  return parts.join("\n")
}

/**
 * Index a single test
 */
export async function indexTest(
  organizationId: number,
  test: {
    testSignature: string
    testName: string
    testFile: string
    testDescription?: string
    executionId?: number
    status?: string
  }
): Promise<{ success: boolean; error?: string }> {
  const sql = getSql()

  try {
    // Prepare metadata and generate embedding
    const metadata = prepareTestMetadata(test.testName, test.testFile, test.testDescription)
    const embedding = await generateEmbedding(metadata)
    const vectorStr = toVectorString(embedding)

    // Upsert into search index
    await sql`
      INSERT INTO test_search_index (
        organization_id,
        test_signature,
        test_name,
        test_file,
        test_description,
        search_embedding,
        last_seen_execution_id,
        total_runs,
        last_status,
        last_indexed_at
      ) VALUES (
        ${organizationId},
        ${test.testSignature},
        ${test.testName},
        ${test.testFile},
        ${test.testDescription || null},
        ${vectorStr}::vector,
        ${test.executionId || null},
        1,
        ${test.status || null},
        NOW()
      )
      ON CONFLICT (organization_id, test_signature)
      DO UPDATE SET
        test_name = EXCLUDED.test_name,
        test_file = EXCLUDED.test_file,
        test_description = COALESCE(EXCLUDED.test_description, test_search_index.test_description),
        search_embedding = EXCLUDED.search_embedding,
        last_seen_execution_id = COALESCE(EXCLUDED.last_seen_execution_id, test_search_index.last_seen_execution_id),
        total_runs = test_search_index.total_runs + 1,
        last_status = COALESCE(EXCLUDED.last_status, test_search_index.last_status),
        last_indexed_at = NOW()
    `

    return { success: true }
  } catch (error) {
    const message = error instanceof Error ? error.message : String(error)
    return { success: false, error: message }
  }
}

/**
 * Batch index multiple tests
 */
export async function indexTestsBatch(
  organizationId: number,
  tests: Array<{
    testSignature: string
    testName: string
    testFile: string
    testDescription?: string
    executionId?: number
    status?: string
  }>
): Promise<{
  total: number
  succeeded: number
  failed: number
}> {
  const sql = getSql()
  const stats = { total: tests.length, succeeded: 0, failed: 0 }

  if (tests.length === 0) return stats

  // Prepare all metadata
  const metadataTexts = tests.map((t) =>
    prepareTestMetadata(t.testName, t.testFile, t.testDescription)
  )

  // Generate embeddings in batch
  const embeddings = await generateEmbeddingsBatch(metadataTexts)

  // Insert/update each test
  for (let i = 0; i < tests.length; i++) {
    const test = tests[i]
    const embedding = embeddings[i]

    if (!embedding || embedding.length !== 768) {
      stats.failed++
      continue
    }

    try {
      const vectorStr = toVectorString(embedding)

      await sql`
        INSERT INTO test_search_index (
          organization_id,
          test_signature,
          test_name,
          test_file,
          test_description,
          search_embedding,
          last_seen_execution_id,
          total_runs,
          last_status,
          last_indexed_at
        ) VALUES (
          ${organizationId},
          ${test.testSignature},
          ${test.testName},
          ${test.testFile},
          ${test.testDescription || null},
          ${vectorStr}::vector,
          ${test.executionId || null},
          1,
          ${test.status || null},
          NOW()
        )
        ON CONFLICT (organization_id, test_signature)
        DO UPDATE SET
          test_name = EXCLUDED.test_name,
          test_file = EXCLUDED.test_file,
          test_description = COALESCE(EXCLUDED.test_description, test_search_index.test_description),
          search_embedding = EXCLUDED.search_embedding,
          last_seen_execution_id = COALESCE(EXCLUDED.last_seen_execution_id, test_search_index.last_seen_execution_id),
          total_runs = test_search_index.total_runs + 1,
          last_status = COALESCE(EXCLUDED.last_status, test_search_index.last_status),
          last_indexed_at = NOW()
      `

      stats.succeeded++
    } catch (error) {
      console.error(`Failed to index test ${test.testSignature}:`, error)
      stats.failed++
    }
  }

  return stats
}

/**
 * Get tests needing indexing
 */
export async function getTestsNeedingIndexing(
  organizationId: number,
  limit: number = 100
): Promise<
  Array<{
    testSignature: string
    testName: string
    testFile: string
  }>
> {
  const sql = getSql()

  // Find tests in test_results that aren't in the search index
  const results = await sql`
    SELECT DISTINCT ON (tr.test_signature)
      tr.test_signature,
      tr.test_name,
      tr.test_file
    FROM test_results tr
    INNER JOIN test_executions te ON tr.execution_id = te.id
    WHERE te.organization_id = ${organizationId}
      AND tr.test_signature IS NOT NULL
      AND NOT EXISTS (
        SELECT 1 FROM test_search_index tsi
        WHERE tsi.organization_id = ${organizationId}
          AND tsi.test_signature = tr.test_signature
          AND tsi.search_embedding IS NOT NULL
      )
    ORDER BY tr.test_signature, tr.created_at DESC
    LIMIT ${limit}
  `

  return results as Array<{
    testSignature: string
    testName: string
    testFile: string
  }>
}

/**
 * Update test stats (pass rate, run count)
 */
export async function updateTestStats(
  organizationId: number
): Promise<number> {
  const sql = getSql()

  // Update stats from test_results
  const result = await sql`
    UPDATE test_search_index tsi
    SET
      total_runs = stats.run_count,
      pass_rate = stats.pass_rate,
      last_status = stats.last_status
    FROM (
      SELECT
        tr.test_signature,
        COUNT(*) as run_count,
        AVG(CASE WHEN tr.status = 'passed' THEN 1.0 ELSE 0.0 END) as pass_rate,
        (
          SELECT status FROM test_results tr2
          WHERE tr2.test_signature = tr.test_signature
          ORDER BY tr2.created_at DESC LIMIT 1
        ) as last_status
      FROM test_results tr
      INNER JOIN test_executions te ON tr.execution_id = te.id
      WHERE te.organization_id = ${organizationId}
        AND tr.test_signature IS NOT NULL
      GROUP BY tr.test_signature
    ) stats
    WHERE tsi.organization_id = ${organizationId}
      AND tsi.test_signature = stats.test_signature
  `

  return result.length
}
```

### Step 5.3: Create Backfill API

**File:** `app/api/admin/index-tests/route.ts`

```typescript
/**
 * Admin API: Index tests for semantic search
 *
 * POST /api/admin/index-tests
 *
 * Indexes test metadata for semantic search capability.
 */

import { NextResponse } from "next/server"
import { getSessionContext } from "@/lib/session-context"
import {
  indexTestsBatch,
  getTestsNeedingIndexing,
  updateTestStats,
} from "@/lib/services/test-indexing-service"

export const dynamic = "force-dynamic"
export const maxDuration = 300 // 5 minutes

export async function POST(request: Request) {
  try {
    const context = await getSessionContext()
    if (!context?.isAdmin) {
      return NextResponse.json(
        { error: "Admin access required" },
        { status: 403 }
      )
    }

    const body = await request.json().catch(() => ({}))
    const limit = Math.min(body.limit || 500, 1000)

    // Get tests needing indexing
    const tests = await getTestsNeedingIndexing(context.organizationId, limit)

    if (tests.length === 0) {
      return NextResponse.json({
        message: "All tests are indexed",
        stats: { total: 0, succeeded: 0, failed: 0 },
      })
    }

    // Index tests
    const stats = await indexTestsBatch(context.organizationId, tests)

    // Update stats
    await updateTestStats(context.organizationId)

    return NextResponse.json({
      message: `Indexed ${stats.succeeded} tests`,
      stats,
    })
  } catch (error) {
    console.error("Index tests error:", error)
    return NextResponse.json(
      { error: "Failed to index tests" },
      { status: 500 }
    )
  }
}

export async function GET() {
  try {
    const context = await getSessionContext()
    if (!context?.isAdmin) {
      return NextResponse.json(
        { error: "Admin access required" },
        { status: 403 }
      )
    }

    const tests = await getTestsNeedingIndexing(context.organizationId, 1)

    // Get counts
    const sql = (await import("@/lib/db/connection")).getSql()

    const [indexed] = await sql`
      SELECT COUNT(*) as count
      FROM test_search_index
      WHERE organization_id = ${context.organizationId}
        AND search_embedding IS NOT NULL
    `

    const [total] = await sql`
      SELECT COUNT(DISTINCT test_signature) as count
      FROM test_results tr
      INNER JOIN test_executions te ON tr.execution_id = te.id
      WHERE te.organization_id = ${context.organizationId}
        AND tr.test_signature IS NOT NULL
    `

    const indexedCount = Number(indexed.count)
    const totalCount = Number(total.count)
    const needsIndexing = totalCount - indexedCount

    return NextResponse.json({
      indexed: indexedCount,
      total: totalCount,
      needsIndexing,
      percentIndexed: totalCount > 0 ? ((indexedCount / totalCount) * 100).toFixed(1) : 0,
    })
  } catch (error) {
    console.error("Check index status error:", error)
    return NextResponse.json(
      { error: "Failed to check index status" },
      { status: 500 }
    )
  }
}
```

import { indexTestsBatch } from "@/lib/services/test-indexing-service"

// After inserting test results, index new tests
// This runs asynchronously

const testsToIndex = testResults.map((tr) => ({
  testSignature: tr.test_signature,
  testName: tr.test_name,
  testFile: tr.test_file,
  executionId: executionId,
  status: tr.status,
}))

// Fire and forget - don't block response
indexTestsBatch(organizationId, testsToIndex)
  .then((stats) => {
    console.log(
      `Indexed ${stats.succeeded}/${stats.total} tests for execution ${executionId}`
    )
  })
  .catch((error) => {
    console.error(`Test indexing failed for execution ${executionId}:`, error)
