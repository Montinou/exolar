```typescript
import { Octokit } from "@octokit/rest";

let octokitInstance: Octokit | null = null;

export function getOctokit(): Octokit {
  if (!octokitInstance) {
    const token = process.env.GITHUB_TOKEN;
    if (!token) {
      throw new Error("GITHUB_TOKEN environment variable is required");
    }
    octokitInstance = new Octokit({ auth: token });
  }
  return octokitInstance;
}

export function parseRepository(repository: string): { owner: string; repo: string } {
  const parts = repository.split("/");
  if (parts.length !== 2) {
    throw new Error(`Invalid repository format: ${repository}. Expected owner/repo`);
  }
  return { owner: parts[0], repo: parts[1] };
}
```

### 3. Create Comment Formatter (`lib/github/comment-formatter.ts`)

```typescript
import { TestExecution } from "../types";

interface CommentData {
  execution: TestExecution;
  dashboard_url: string;
  flaky_count?: number;
}

export function formatPRComment(data: CommentData): string {
  const { execution, dashboard_url, flaky_count = 0 } = data;
  const isSuccess = execution.status === "success";
  const statusEmoji = isSuccess ? "✅" : "❌";
  const statusText = isSuccess ? "Passed" : "Failed";

  const totalTests = execution.passed + execution.failed + execution.skipped;
  const passRate = totalTests > 0
    ? ((execution.passed / totalTests) * 100).toFixed(1)
    : "0";

  const duration = execution.duration_ms
    ? execution.duration_ms < 60000
      ? `${(execution.duration_ms / 1000).toFixed(1)}s`
      : `${(execution.duration_ms / 60000).toFixed(1)}m`
    : "N/A";

  let comment = `## ${statusEmoji} E2E Tests ${statusText}

| Metric | Value |
|--------|-------|
| **Status** | ${statusText} |
| **Pass Rate** | ${passRate}% |
| **Duration** | ${duration} |

### Results

| Status | Count |
|--------|-------|
| ✅ Passed | ${execution.passed} |
| ❌ Failed | ${execution.failed} |
| ⏭️ Skipped | ${execution.skipped} |
`;

  if (flaky_count > 0) {
    comment += `| ⚠️ Flaky | ${flaky_count} |\n`;
  }

  comment += `
### Details

- **Branch:** \`${execution.branch}\`
- **Commit:** \`${execution.commit_sha.substring(0, 7)}\`
- **Run ID:** \`${execution.run_id}\`

[📊 View Full Report](${dashboard_url})

---
<sub>🤖 Generated by E2E Test Dashboard</sub>`;

  return comment;
}

// Signature to identify our comments for updates
export const COMMENT_SIGNATURE = "<!-- e2e-test-dashboard-comment -->";

export function formatPRCommentWithSignature(data: CommentData): string {
  return `${COMMENT_SIGNATURE}\n${formatPRComment(data)}`;
}
```

### 4. Create GitHub PR Service (`lib/github/pr-comments.ts`)

```typescript
import { getOctokit, parseRepository } from "./client";
import { formatPRCommentWithSignature, COMMENT_SIGNATURE } from "./comment-formatter";
import { TestExecution } from "../types";

interface PostCommentOptions {
  execution: TestExecution;
  repository: string;
  prNumber: number;
  dashboardUrl: string;
  flakyCount?: number;
  updateExisting?: boolean;
}

export async function postPRComment(
  options: PostCommentOptions
): Promise<{ success: boolean; commentId?: number; error?: string }> {
  const {
    execution,
    repository,
    prNumber,
    dashboardUrl,
    flakyCount = 0,
    updateExisting = true,
  } = options;

  try {
    const octokit = getOctokit();
    const { owner, repo } = parseRepository(repository);

    const body = formatPRCommentWithSignature({
      execution,
      dashboard_url: dashboardUrl,
      flaky_count: flakyCount,
    });

    // Check for existing comment to update
    if (updateExisting) {
      const existingComment = await findExistingComment(owner, repo, prNumber);
      if (existingComment) {
        await octokit.issues.updateComment({
          owner,
          repo,
          comment_id: existingComment.id,
          body,
        });
        return { success: true, commentId: existingComment.id };
      }
    }

    // Create new comment
    const response = await octokit.issues.createComment({
      owner,
      repo,
      issue_number: prNumber,
      body,
    });

    return { success: true, commentId: response.data.id };
  } catch (error) {
    console.error("Error posting PR comment:", error);
    return {
      success: false,
      error: error instanceof Error ? error.message : "Unknown error",
    };
  }
}

async function findExistingComment(
  owner: string,
  repo: string,
  prNumber: number
): Promise<{ id: number } | null> {
  try {
    const octokit = getOctokit();

    const { data: comments } = await octokit.issues.listComments({
      owner,
      repo,
      issue_number: prNumber,
      per_page: 100,
    });

    const existingComment = comments.find((comment) =>
      comment.body?.includes(COMMENT_SIGNATURE)
    );

    return existingComment ? { id: existingComment.id } : null;
  } catch {
    return null;
  }
}

// Delete PR comment
export async function deletePRComment(
  repository: string,
  commentId: number
): Promise<boolean> {
  try {
    const octokit = getOctokit();
    const { owner, repo } = parseRepository(repository);

    await octokit.issues.deleteComment({
      owner,
      repo,
      comment_id: commentId,
    });

    return true;
  } catch {
    return false;
  }
}
```

```typescript
// Update execution with PR info
export async function updateExecutionPRInfo(
  executionId: number,
  prNumber: number,
  prTitle?: string,
  repository?: string,
  commentId?: number
): Promise<void> {
  const sql = getSql();

  await sql`
    UPDATE test_executions
    SET
      pr_number = ${prNumber},
      pr_title = ${prTitle || null},
      repository = ${repository || null},
      github_comment_id = ${commentId || null}
    WHERE id = ${executionId}
  `;
}

// Get execution by PR
export async function getExecutionsByPR(
  repository: string,
  prNumber: number,
  limit: number = 10
): Promise<TestExecution[]> {
  const sql = getSql();

  const results = await sql`
    SELECT *
    FROM test_executions
    WHERE repository = ${repository}
      AND pr_number = ${prNumber}
    ORDER BY started_at DESC
    LIMIT ${limit}
  `;

  return results as TestExecution[];
}
```

### 6. Create GitHub API Routes

#### `app/api/github/comment/route.ts`

```typescript
import { postPRComment } from "@/lib/github/pr-comments";
import { getExecutionById, updateExecutionPRInfo } from "@/lib/db";
import { NextRequest, NextResponse } from "next/server";

export async function POST(request: NextRequest) {
  try {
    const body = await request.json();
    const {
      execution_id,
      repository,
      pr_number,
      pr_title,
      flaky_count = 0,
    } = body;

    if (!execution_id || !repository || !pr_number) {
      return NextResponse.json(
        { error: "execution_id, repository, and pr_number are required" },
        { status: 400 }
      );
    }

    const execution = await getExecutionById(execution_id);
    if (!execution) {
      return NextResponse.json(
        { error: "Execution not found" },
        { status: 404 }
      );
    }

    const dashboardUrl = `${process.env.NEXT_PUBLIC_APP_URL}/executions/${execution_id}`;

    const result = await postPRComment({
      execution,
      repository,
      prNumber: pr_number,
      dashboardUrl,
      flakyCount: flaky_count,
      updateExisting: true,
    });

    if (result.success) {
      // Update execution with PR info
      await updateExecutionPRInfo(
        execution_id,
        pr_number,
        pr_title,
        repository,
        result.commentId
      );
    }

    return NextResponse.json(result);
  } catch (error) {
    console.error("Error posting GitHub comment:", error);
    return NextResponse.json(
      { error: "Failed to post comment" },
      { status: 500 }
    );
  }
}
```

## GitHub Actions Integration

### Example Workflow Step

```yaml
# .github/workflows/e2e-tests.yml
- name: Post Test Results to PR
  if: github.event_name == 'pull_request'
  env:
    DASHBOARD_API_URL: ${{ secrets.DASHBOARD_API_URL }}
    DASHBOARD_API_KEY: ${{ secrets.DASHBOARD_API_KEY }}
  run: |
    curl -X POST "$DASHBOARD_API_URL/api/github/comment" \
      -H "Content-Type: application/json" \
      -H "Authorization: Bearer $DASHBOARD_API_KEY" \
      -d '{
        "execution_id": ${{ steps.upload_results.outputs.execution_id }},
        "repository": "${{ github.repository }}",
        "pr_number": ${{ github.event.pull_request.number }},
        "pr_title": "${{ github.event.pull_request.title }}",
        "flaky_count": ${{ steps.test.outputs.flaky_count || 0 }}
      }'
```
